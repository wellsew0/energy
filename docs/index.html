making a plot

library(ggplot2)
ggplot(econ, 
       aes(x=year,
           y= cost,
           color=source)) + 
  geom_point() +
  geom_hline(yintercept=0.05, lty=2) + 
  geom_hline(yintercept=0.18, lty=3) +
  annotate(geom='text',
           x=1990, y=0.1,
           size=5,
           label='Range of fossil\n fuel costs')
